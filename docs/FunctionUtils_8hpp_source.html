24  //! Numerically approximates the derivative of a single-variable function

27  //! \return the derivative of the function
28  static double derivative(const function<double(double)> &f, double x) {

30  return (f(x + h) - f(x)) / h;

33  //! Numerically approximates the partial derivative of a two-dimensional function
34  //! \param f a function
35  //! \param x the first point at which the derivative is to be calculated
36  //! \param y the second point at which the derivative is to be calculated
37  //! \param which 0 to differentiate x, otherwise y
38  //! \return the partial derivative of the function
39  static double partialDerivative(const function<double(double, double)> &f, double x,
40  double y, int which = 0) {
41  double h = sqrtMachineEpsilon;
42  if (which == 0) {
43  h *= x;
44  return (f(x, y) - f(x - h, y)) / h;
45  }
46  h *= y;
47  return (f(x, y) - f(x, y - h)) / h;
48  }
