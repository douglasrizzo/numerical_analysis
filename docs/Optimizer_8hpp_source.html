53  //! Adaptive quadrature recursive method
54  //! \param f function to integrate
55  //! \param a the lower bound of the integration interval
56  //! \param b the upper bound of the integration interval
57  //! \param method the Newton-Cotes function to use in the approximation
58  //! \param error
59  //! \return Numerical approximation of the integral of f
60  double innerAdaptiveIntegration(const function<double(double)> &f, double a,
61  double b, IntegrationMethod method,
62  double error = 1e-12) {
63  iterations += 2;
64  // calculates the middle point between a and b
65  double meio = (b + a) / 2;
66  // uses the integrate method to calculate the value of a single quadrature
67  // vs. the sum of two sub-quadratures
68  double i1 = integrate(f, a, b, 1, method),
69  i2 = integrate(f, a, meio, 1, method) +
70  integrate(f, meio, b, 1, method);
71 
72  // if there is error, run adaptive integration in the two sub-divisions of
73  // the current partition
74  if (fabs(i1 - i2) > error) {
75  return innerAdaptiveIntegration(f, a, meio, method, error) +
76  innerAdaptiveIntegration(f, meio, b, method, error);
77  }
78 
79  // otherwise, return the most precise value of the two already calculated
80  return i2;
81  }

109  //! Numerically searches for the root of a function via Newton-Raphson method
110  //! \param f a function
111  //! \param x the initial point to start the search
112  //! \param error minimum tolerance for the search to end
113  //! \param max_iters maximum number of iterations
114  //! \param learnRate the learning rate of the search
115  //! \param verbose whether to print a short summary of the search at every iteration
116  //! \return the point at which the function intercepts the x-axis
117  double findRoot(const function<double(double)> &f, double x,
118  double error = 1e-8, int max_iters = 1000,
119  double learnRate = 1, bool verbose = false) throw(runtime_error) {
120  endReason = "You didn't run any optimization yet!";
121  iterations = 0;
122  double f_val;
123 
124  auto start = clock::now();
125  while (true) {
126  f_val = f(x);
127  double aux = x + learnRate * - f_val / FunctionUtils::derivative(f, x);
128  if (aux == x) {
129  this->endReason = "No change in x from previous iteration";
130  break;
131  }
132 
133  iterations ++;
134  x = aux;
135  if (verbose) {
136  cout << "Iteration " << iterations << ": x = " << x
137  << ", f(x) = " << f_val << '\n';
138  }
139 
140  if (iterations >= max_iters) {
141  string message = "Maximum number of iterations reached";
142  this->endReason = message;
143  throw runtime_error(message);
144  }
145  if (fabs(f_val) < error) {
146  this->endReason = "Minimum error threshold reached";
147  break;
148  }
149  }
150  endClock(start);

261  //! Numerically approximates the integral of a function
262  //! \param f the function to integrate
263  //! \param low the lower bound of the integration interval
264  //! \param high the upper bound of the integration interval
265  //! \param points the number of quadrature points to use in the approximation
266  //! \param method the Newton-Cotes function to use in the approximation
267  //! \return Numerical approximation of the integral of f
268  double integrate(const function<double(double)> &f, double low,
269  double high, long int points = 40,
270  IntegrationMethod method = SIMPSON) throw(runtime_error) {
271 
272  if (low == high) {
273  throw runtime_error("Lower bound of integration = Higher bound");
274  }
275  if (low > high) {
276  double temp = low;
277  low = high;
278  high = temp;
279  }
280 
281  function<double(function<double(double)>, double, double)> approx;
282 
283  if (method == SIMPSON)
284  approx = FunctionUtils::simpsonRule;
285  else if (method == RECTANGLE)
286  approx = FunctionUtils::rectangleRule;
287  else if (method == TRAPEZOID)
288  approx = FunctionUtils::trapezoidRule;
289  else
290  throw runtime_error("Unsupported integration method");
291 
292  auto start = clock::now();
293 
294  double sum = 0;
295  double step = (high - low) / points;
296 
297  if (step < 1e-8)
298  throw runtime_error("Step size of " + to_string(step) + " is too small to be precise");
299 
300 #pragma omp parallel for reduction(+:sum)
301  for (int i = 0; i < points; i ++)
302  sum += approx(f, low + (i * step), low + ((i + 1) * step));
303 
304  endClock(start);
305  return sum;
306  }

349  VolumousObject monteCarloVolume(double xLow,
350  double xHigh,
351  double yLow,
352  double yHigh,
353  double zLow,
354  double zHigh,
355  const function<bool(double, double, double)> &isInside,
356  long int points) {
357  VolumousObject obj;
358  // number of pts inside the object
359  int pointsInside = 0;
360  // sum of the x, y and z coordinates of the pts inside the object
361  // useful for center of mass later
362  double xSum = 0, ySum = 0, zSum = 0;
363 
364  auto start = clock::now();
365 
366 #pragma omp parallel for reduction(+:xSum) reduction(+:ySum) reduction(+:zSum) reduction(+:pointsInside)
367  for (long int i = 0; i < points; i ++) {
368  double x = myRandom(xLow, xHigh);
369  double y = myRandom(yLow, yHigh);
370  double z = myRandom(zLow, zHigh);
371 
372  if (isInside(x, y, z)) {
373  pointsInside ++;
374  xSum += x;
375  ySum += y;
376  zSum += z;
377  }
378  }
379 
380  double pctInside = pointsInside / (double) points;
381 
382  // volume of the enclosing cube
383  double cubeVolume = (xHigh - xLow) * (yHigh - yLow) * (zHigh - zLow);
384  // estimated volume of our object
385  obj.setVolume(cubeVolume * pctInside);
386  //weight due to gravity
387  obj.setWeight(obj.getVolume());
388 
389  // according to Numerical Recipes, a suitable error measure is +/- 1 standard deviation
390  error = cubeVolume * sqrt((pctInside - pow(pctInside, 2.0f)) / points);
391  obj.setError(error);
392 
393  // center of mass in the three coordinates
394  // density = 1 everywhere, so it's a straightforward sum
395  obj.getCenterOfMass().setX(xSum / pointsInside);
396  obj.getCenterOfMass().setY(ySum / pointsInside);
397  obj.getCenterOfMass().setZ(zSum / pointsInside);
398 
399  endClock(start);
400 
401  return obj;
402  }
