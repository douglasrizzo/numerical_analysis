37  // calculates the middle point between a and b

49  innerAdaptiveIntegration(f, meio, b, method, error);

51

242

245

248 
249  for (int i = 0; i < points; i ++)
250  sum += approx(f, low + (i * step), low + ((i + 1) * step));
251 
252  return sum;
